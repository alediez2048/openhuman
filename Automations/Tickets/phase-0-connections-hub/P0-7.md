    4. **Delete the connection.**

## Suggested Implementation Pattern

**`app/test/e2e/specs/connections-hub.spec.ts` (extract):**

```ts
import { expect } from 'chai';
import {
  waitForWebView,
  clickNativeButton,
  setNativeInputValue,
  assertVisibleByTestId,
} from '../helpers/element-helpers';
import { resetMockBackend, getMockRequests } from '../helpers/mock-helpers';

describe('Connections Hub — Phase 0', () => {
  before(async () => {
    await resetMockBackend();
    await waitForWebView();
  });

  it('renders all 6 sections on a fresh workspace', async () => {
    await clickNativeButton('bottom-tab-connections');
    await waitForWebView();
    await assertVisibleByTestId('connections-page-root');
    for (const section of ['composio', 'channels', 'webview', 'builtin', 'mcp', 'http']) {
      await assertVisibleByTestId(`connections-section-${section}`);
    }
  });

  it('creates a Generic HTTP connection', async () => {
    await clickNativeButton('connections-generic-http-add');
    await setNativeInputValue('http-modal-name', 'e2e-test-conn');
    await setNativeInputValue('http-modal-base-url', 'http://127.0.0.1:8765/echo');
    await clickNativeButton('http-modal-auth-bearer');
    await setNativeInputValue('http-modal-credential', 'e2e-token');
    await clickNativeButton('http-modal-save');
    await waitForWebView();
    await assertVisibleByTestId('http-card-e2e-test-conn');
    const requests = await getMockRequests();
    expect(requests.find(r => r.method === 'connections_generic_http_create')).to.exist;
  });

  it('tests the connection', async () => {
    await clickNativeButton('http-card-e2e-test-conn-test');
    await waitForWebView();
    await assertVisibleByTestId('toast-success');
  });

  it('deletes the connection', async () => {
    await clickNativeButton('http-card-e2e-test-conn-delete');
    await clickNativeButton('confirm-delete');
    await waitForWebView();
    await assertNotVisibleByTestId('http-card-e2e-test-conn');
  });

  it('redirects /skills to /connections', async () => {
    await browser.url('app://./#/skills');
    await waitForWebView();
    const url = await browser.getUrl();
    expect(url).to.match(/#\/connections$/);
  });

  it('redirects /channels to /connections#channels', async () => {
    await browser.url('app://./#/channels');
    await waitForWebView();
    const url = await browser.getUrl();
    expect(url).to.match(/#\/connections#channels$/);
    await assertVisibleByTestId('connections-section-channels');
  });
});
```

## Edge Cases to Handle

1. **E2E spec on Linux vs macOS.** Driver differences — Appium Mac2 uses `app://./#/...` URL navigation; tauri-driver may need `browser.url('http://tauri.localhost/#/...')`. Use the existing `helpers/navigation.ts` (if present) to abstract.
2. **Bottom-tab id changed in P0-4 from `'skills'` to `'connections'`.** The test ID for the tab should be `bottom-tab-connections`, not `bottom-tab-skills`. Verify in `BottomTabBar.tsx`.
3. **The Channels section deep-link scroll behavior.** If the scroll-into-view is implemented via `useEffect` after section mount, the spec may need a small `waitForWebView()` after the URL navigation before asserting visibility.
4. **Mock backend doesn't respond to HEAD on `/echo`.** Verify `scripts/mock-api-core.mjs` handles HEAD; if not, extend it to respond 200 to HEAD on any registered path. (This is the only mock-backend change permitted in P0-7.)
5. **The Generic HTTP "Test" button toast appears and disappears quickly.** Use a `waitFor` with a short retry rather than a strict `assertVisible` snapshot.
6. **`/skills` and `/channels` redirects on HashRouter.** Confirm the URL assertion matches HashRouter format (`#/connections`, not just `/connections`).
7. **Capability catalog ordering.** Insert in the right place (chronological/phase order). Don't break tests asserting on neighbor ordering.
8. **CI flakiness on the search debounce.** The search input has a 200ms debounce — `waitForWebView` may not capture it. Use a small explicit `browser.pause(300)` after typing or assert with a longer timeout.
9. **DEVLOG summary mentions PR numbers that don't exist yet.** Use placeholder `#TBD` and let the merging contributor (or follow-up agent) replace them.
10. **`about_app` consumers cache the catalog.** Verify the new entry is visible without a core restart — the catalog is typically loaded eagerly at startup, so no special handling.
